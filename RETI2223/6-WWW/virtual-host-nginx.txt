Ho modificato il file /etc/nginx/sites-available/default:

    server {
        listen 81;
        listen [::]:81;

        server_name vh1;

        root /var/www/vh1;
        index index.html;

        location / {
                try_files $uri $uri/ =404;
        }
    }
