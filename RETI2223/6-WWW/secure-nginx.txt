Installazione:
- sudo apt install apache2-utils

Configurazione:
- root@salvatale-server:/var/www/vh1# htpasswd -c /etc/nginx/.htpasswd ubuntu
	New password:
	Re-type new password:
	Adding password for user ubuntu

- modificare il file /etc/nginx/sites-available/default" :
	
	location /secure {
		try_files $uri $uri/ =404;
		auth_basic "Secure area";
		auth_basic_user_file /etc/nginx/.htpasswd;
        }

Ho creato la directory secure : sudo mkdir -p /var/www/html/secure

root@salvatale-server:~# service nginx restart
